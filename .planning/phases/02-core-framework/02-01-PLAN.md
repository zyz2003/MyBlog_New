### Task 01-01-03: 实现 HookRegistry（挂载点注册表）

**Type:** implementation
**Requirements:** PLUGIN-02

实现类型安全的挂载点注册表：

```typescript
import { HookName, Hook, SyncHook, AsyncHook } from './types'

// 每个挂载点的上下文类型映射
export interface HookContextMap {
  [HookName.APP_INIT]: { app: App }
  [HookName.APP_CREATED]: { app: App; nuxt: any }
  [HookName.APP_MOUNTED]: { app: App; el: Element }
  [HookName.ROUTER_BEFORE_EACH]: { to: any; from: any; next: () => void }
  [HookName.ROUTER_AFTER_EACH]: { to: any; from: any }
  [HookName.PAGE_LOADING_START]: { url: string }
  [HookName.PAGE_LOADING_END]: { url: string }
  [HookName.PAGE_RENDER]: { html: string }
  [HookName.COMPONENT_MOUNT]: { component: any; props: any }
  [HookName.COMPONENT_UNMOUNT]: { component: any }
  [HookName.API_REQUEST]: { url: string; options: RequestInit }
  [HookName.API_RESPONSE]: { url: string; response: Response }
  [HookName.API_ERROR]: { url: string; error: Error }
  [HookName.THEME_CHANGE]: { theme: string }
  [HookName.STORAGE_SET]: { key: string; value: any }
  [HookName.STORAGE_GET]: { key: string }
}

export class HookRegistry {
  private hooks: Map<HookName, Hook[]> = new Map()

  /**
   * 注册挂载点
   */
  register<T extends HookName>(
    hookName: T,
    fn: (ctx: HookContextMap[T]) => HookContextMap[T] | Promise<HookContextMap[T]>,
    type: 'sync' | 'async' = 'async'
  ): void {
    const hook: Hook = type === 'sync'
      ? { type: 'sync', fn: fn as (ctx: any) => any }
      : { type: 'async', fn: fn as (ctx: any) => Promise<any> }

    if (!this.hooks.has(hookName)) {
      this.hooks.set(hookName, [])
    }
    this.hooks.get(hookName)!.push(hook)
  }

  /**
   * 执行同步挂载点
   */
  async executeSync<T extends HookName>(
    hookName: T,
    context: HookContextMap[T]
  ): Promise<HookContextMap[T]> {
    const hooks = this.hooks.get(hookName) || []
    let ctx = context

    for (const hook of hooks) {
      if (hook.type === 'sync') {
        ctx = hook.fn(ctx) as HookContextMap[T]
      } else {
        ctx = await hook.fn(ctx) as HookContextMap[T]
      }
    }

    return ctx
  }

  /**
   * 执行异步挂载点（并行）
   */
  async executeParallel<T extends HookName>(
    hookName: T,
    context: HookContextMap[T]
  ): Promise<void> {
    const hooks = this.hooks.get(hookName) || []
    const promises = hooks.map(hook => hook.fn(context))
    await Promise.all(promises)
  }

  /**
   * 获取挂载点数量
   */
  getCount(hookName: HookName): number {
    return (this.hooks.get(hookName) || []).length
  }

  /**
   * 清空所有挂载点
   */
  clear(): void {
    this.hooks.clear()
  }
}
```

// packages/plugin/src/lifecycle/events.ts
export enum LifecycleEvent {
  INIT = 'plugin:init',
  REGISTER = 'plugin:register',
  APP_CREATED = 'app:created',
  APP_MOUNT = 'app:mounted',
  NAVIGATION_START = 'navigation:start',
  NAVIGATION_END = 'navigation:end',
  RENDER = 'render',
  DESTROY = 'plugin:destroy'
}

export type LifecycleCallback = (data?: any) => void | Promise<void>

export class LifecycleEmitter {
  private listeners: Map<LifecycleEvent, LifecycleCallback[]> = new Map()

  on(event: LifecycleEvent, callback: LifecycleCallback): void {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, [])
    }
    this.listeners.get(event)!.push(callback)
  }

  async emit(event: LifecycleEvent, data?: any): Promise<void> {
    const callbacks = this.listeners.get(event) || []
    for (const callback of callbacks) {
      await callback(data)
    }
  }

  off(event: LifecycleEvent, callback?: LifecycleCallback): void {
    if (!callback) {
      this.listeners.delete(event)
    } else {
      const callbacks = this.listeners.get(event) || []
      const index = callbacks.indexOf(callback)
      if (index > -1) {
        callbacks.splice(index, 1)
      }
    }
  }
}

### Task 01-01-05: 实现 PluginManager（插件管理器）

**Type:** implementation
**Requirements:** PLUGIN-01, PLUGIN-02, PLUGIN-03

实现核心插件管理器：

```typescript
import { Plugin, PluginContext, PluginConfig, PluginManagerOptions } from './types'
import { HookRegistry } from './HookRegistry'
import { LifecycleEmitter, LifecycleEvent } from './lifecycle/events'

export class PluginManager {
  private plugins: Map<string, Plugin> = new Map()
  private hooks: HookRegistry
  private emitter: LifecycleEmitter
  private config: PluginConfig[] = []

  constructor(options: PluginManagerOptions = {}) {
    this.hooks = new HookRegistry()
    this.emitter = new LifecycleEmitter()

    if (options.configPath) {
      this.loadConfig(options.configPath)
    }
  }

  /**
   * 加载插件配置
   */
  async loadConfig(configPath: string): Promise<void> {
    try {
      const config = await import(configPath)
      this.config = config.plugins || []
      await this.emitter.emit(LifecycleEvent.REGISTER, { plugins: this.config })
    } catch (error) {
      console.warn(`Failed to load config from ${configPath}:`, error)
    }
  }

  /**
   * 注册插件
   */
  register(plugin: Plugin): void {
    if (this.plugins.has(plugin.name)) {
      console.warn(`Plugin "${plugin.name}" is already registered`)
      return
    }
    this.plugins.set(plugin.name, plugin)
  }

  /**
   * 加载并注册插件
   */
  async loadPlugin(pluginPath: string): Promise<void> {
    try {
      const module = await import(pluginPath)
      const plugin: Plugin = module.default || module
      this.register(plugin)

      // 触发初始化钩子
      if (plugin.onInit) {
        const ctx: PluginContext = {
          app: this.getApp(),
          config: this.getPluginConfig(plugin.name),
          hooks: this.hooks
        }
        await plugin.onInit(ctx)
      }
    } catch (error) {
      console.error(`Failed to load plugin from ${pluginPath}:`, error)
    }
  }

  /**
   * 卸载插件
   */
  async unregister(pluginName: string): Promise<void> {
    const plugin = this.plugins.get(pluginName)
    if (plugin) {
      if (plugin.onDestroy) {
        await plugin.onDestroy()
      }
      this.plugins.delete(pluginName)
    }
  }

  /**
   * 获取插件
   */
  getPlugin(name: string): Plugin | undefined {
    return this.plugins.get(name)
  }

  /**
   * 获取所有已注册插件
   */
  getPlugins(): Plugin[] {
    return Array.from(this.plugins.values())
  }

  /**
   * 获取挂载点注册表
   */
  get hooks(): HookRegistry {
    return this._hooks
  }

  /**
   * 获取生命周期发射器
   */
  get emitter(): LifecycleEmitter {
    return this._emitter
  }

  // 私有方法
  private getApp(): App<any> {
    // 在实际实现中，这里会返回 Nuxt app 实例
    return {} as App<any>
  }

  private getPluginConfig(pluginName: string): Record<string, any> {
    const config = this.config.find(p => p.name === pluginName)
    return config?.config || {}
  }

  private _hooks: HookRegistry
  private _emitter: LifecycleEmitter
}
```
