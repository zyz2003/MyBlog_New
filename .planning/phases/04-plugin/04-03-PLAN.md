**Pinia store** (`stores/plugin.ts`):
```typescript
import { defineStore } from 'pinia'

interface PluginInfo {
  name: string
  label: string
  type: string
  mountPoints: string[]
  enabled: boolean
  config: Record<string, any>
}

export const usePluginStore = defineStore('plugin', () => {
  const plugins = ref<PluginInfo[]>([])
  const loading = ref(false)

  async function fetchPlugins() {
    loading.value = true
    try {
      const { data } = await $fetch('/api/plugins')
      if (data) plugins.value = data
    } finally {
      loading.value = false
    }
  }

  async function enablePlugin(name: string, config: Record<string, any>) {
    await $fetch(`/api/plugins/${name}/enable`, { method: 'POST', body: { config } })
    await fetchPlugins()
  }

  async function disablePlugin(name: string) {
    await $fetch(`/api/plugins/${name}/disable`, { method: 'POST' })
    await fetchPlugins()
  }

  async function updateConfig(name: string, config: Record<string, any>) {
    await $fetch(`/api/plugins/${name}/config`, { method: 'PUT', body: { config } })
    await fetchPlugins()
  }

  const enabledPlugins = computed(() => plugins.value.filter(p => p.enabled))

  return { plugins, loading, enabledPlugins, fetchPlugins, enablePlugin, disablePlugin, updateConfig }
})
```

```typescript
import { usePluginStore } from '~/stores/plugin'

export function usePlugin() {
  const store = usePluginStore()

  // Fetch plugins on first use (client-side)
  if (import.meta.client && store.plugins.length === 0) {
    store.fetchPlugins()
  }

  /** Get enabled plugins for a specific mount point */
  function getPluginsForMountPoint(mountPoint: string) {
    return computed(() =>
      store.enabledPlugins.filter(p => p.mountPoints.includes(mountPoint))
    )
  }

  /** Check if a specific plugin is enabled */
  function isPluginEnabled(name: string) {
    return computed(() => store.enabledPlugins.some(p => p.name === name))
  }

  /** Get plugin config by name */
  function getPluginConfig(name: string) {
    return computed(() => store.plugins.find(p => p.name === name)?.config ?? {})
  }

  return {
    plugins: computed(() => store.plugins),
    enabledPlugins: computed(() => store.enabledPlugins),
    loading: computed(() => store.loading),
    getPluginsForMountPoint,
    isPluginEnabled,
    getPluginConfig,
    refresh: store.fetchPlugins,
    enable: store.enablePlugin,
    disable: store.disablePlugin,
    updateConfig: store.updateConfig,
  }
}
```

The composable wraps the Pinia store with convenience methods and computed refs for template reactivity.
