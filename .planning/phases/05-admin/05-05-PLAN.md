```typescript
const uploadFile = async (file: File, folderId?: string) => {
  const formData = new FormData()
  formData.append('file', file)
  if (folderId) formData.append('folderId', folderId)

  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest()

    xhr.upload.addEventListener('progress', (e) => {
      if (e.lengthComputable) {
        const progress = Math.round((e.loaded / e.total) * 100)
        updateProgress(file.name, progress)
      }
    })

    xhr.addEventListener('load', () => {
      if (xhr.status === 200) {
        resolve(JSON.parse(xhr.responseText))
      } else {
        reject(new Error('Upload failed'))
      }
    })

    xhr.addEventListener('error', () => reject(new Error('Upload error')))

    xhr.open('POST', '/api/v1/media')
    xhr.setRequestHeader('Authorization', `Bearer ${token}`)
    xhr.send(formData)
  })
}
```

### Task 6: 创建媒体库 Composable

**文件：** `apps/site/composables/useMediaLibrary.ts`

**功能：**
- 媒体列表获取（带分页、筛选、搜索）

- 错误处理

**代码片段：**
```typescript
import type { Media } from '@my-blog/database'

interface UseMediaLibraryOptions {
  folderId?: string | null
  filterType?: string
  search?: string
}

export function useMediaLibrary() {
  const fetchMedia = async (options: UseMediaLibraryOptions = {}) => {
    const query = new URLSearchParams()
    if (options.folderId) query.append('folder', options.folderId)
    if (options.filterType) query.append('mimeType', options.filterType)
    if (options.search) query.append('search', options.search)
    query.append('limit', '20')
    query.append('offset', '0')

    return await $fetch(`/api/v1/media?${query.toString()}`)
  }

  const uploadMedia = async (
    files: File[],
    folderId?: string,
    onProgress?: (filename: string, progress: number) => void
  ): Promise<Media[]> => {
    const results: Media[] = []
    for (const file of files) {
      const result = await uploadSingleFile(file, folderId, onProgress)
      results.push(result)
    }
    return results
  }

  const deleteMedia = async (id: string) => {
    return await $fetch(`/api/v1/media/${id}`, { method: 'DELETE' })
  }

  const fetchFolders = async () => {
    return await $fetch('/api/v1/media/folders')
  }

  return {
    fetchMedia,
    uploadMedia,
    deleteMedia,
    fetchFolders
  }
}
```

- 选中状态明显高亮
- 确认按钮显示已选图片数量

**使用场景：**
- 文章编辑页封面图选择
- 文章编辑器插入图片

**代码片段：**
```vue
<script setup lang="ts">
const props = defineProps<{
  open: boolean
  multiple?: boolean
}>()

const emit = defineEmits<{
  'update:open': [value: boolean]
  'select': [urls: string[]]
}>()

const selectedMedia = ref<string[]>([])

const handleConfirm = () => {
  emit('select', selectedMedia.value)
  emit('update:open', false)
}
</script>

<template>
  <Dialog :open="props.open" @update:open="emit('update:open', $event)">
    <DialogContent class="max-w-4xl">
      <div class="flex h-[600px]">
        <FolderTree />
        <div class="flex-1 overflow-auto">
          <MediaGrid
            v-model:selected="selectedMedia"
            :multiple="props.multiple"
            :selectable="true"
          />
        </div>
      </div>
      <div class="flex justify-end gap-2">
        <Button @click="handleConfirm">
          插入{{ selectedMedia.length }}张图片
        </Button>
      </div>
    </DialogContent>
  </Dialog>
</template>
```
