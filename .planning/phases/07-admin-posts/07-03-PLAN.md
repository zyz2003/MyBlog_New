     import Vditor from 'vditor'
     import 'vditor/dist/index.css'
     const vditor = new Vditor('vditor-editor', {
       mode: 'ir',  // instant rendering mode
       theme: 'classic',
       toolbar: ['emoji', 'headings', 'bold', 'italic', 'strike', '|', 'line', 'quote', 'list', 'ordered-list', 'check', '|', 'code', 'inline-code', 'table', '|', 'link', 'upload', '|', 'undo', 'redo', '|', 'edit-mode', 'fullscreen', 'preview'],
